1) Correr un modelo (sin scripts)
#Por CLI:
python -m hsi_lab.train --models cnn_baseline --trials 40 --epochs 60

#Con variable de entorno (útil en screen):
export HSI_MODELS="cnn_residual"
python -m hsi_lab.train

2) Correr varios modelos seguidos:
python -m hsi_lab.train --models cnn_baseline,dnn_wide,dnn_selu --trials 30 --epochs 50

3) Guardar el log:
python -m hsi_lab.train --models cnn_baseline --trials 40 --epochs 60 | tee outputs/other_outputs/cnn_baseline_$(date +%Y%m%d-%H%M%S).log

4) Lanzarlo en background con screen:
screen -S exp_cnn -dm bash -lc 'python -m hsi_lab.train --models cnn_baseline --trials 40 --epochs 60 | tee outputs/other_outputs/cnn_baseline_$(date +%Y%m%d-%H%M%S).log'
screen -ls        # ver sesiones
screen -r exp_cnn # entrar a la sesión
# Ctrl+A, luego D para salir dejando el proceso corriendo
